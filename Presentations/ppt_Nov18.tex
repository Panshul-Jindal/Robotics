\section{Introduction}

\begin{frame}{Project Overview}
\begin{itemize}
    \item \textbf{Objective:} Compare path planning and trajectory planning algorithms for PUMA 560 manipulator
    \item \textbf{Path Planning:} Determines waypoints in Cartesian space (where to go)
    \item \textbf{Trajectory Planning:} Generates smooth joint-space motion (how to get there)
    \item \textbf{Approach:} 
    \begin{itemize}
        \item Fix one algorithm type, vary the other
        \item Analyze performance differences
    \end{itemize}
\end{itemize}
\end{frame}

\subsection{Linear Path + Bang-Bang}

\begin{frame}[fragile]{Linear Path + Bang-Bang Trajectory}
\begin{columns}
\column{0.5\textwidth}
\underline{\textbf{Path Planning: Linear}}
\begin{algorithmic}[1]
\scriptsize
\Require Start $S$, End $E$, Points $N$
\For{$i = 0$ to $N-1$}
    \State $t \gets \frac{i}{N-1}$
    \State $P[i] \gets (1-t)S + tE$
\EndFor
\State \Return $P$
\end{algorithmic}

\vspace{0.3cm}
\underline{\textbf{Trajectory: Bang-Bang}}
\begin{algorithmic}[1]
\scriptsize
\Require $q_0, q_1, T$
\If{$\tau < 0.5$}
    \State $q = q_0 + 2(q_1-q_0)\tau^2$
\Else
    \State $q = q_1 - 2(q_1-q_0)(1-\tau)^2$
\EndIf
\end{algorithmic}

\subsection{RRT Path + Bang-Bang}

\begin{frame}[fragile]{RRT Path + Bang-Bang Trajectory}
\begin{columns}
\column{0.5\textwidth}
\underline{\textbf{Path Planning: RRT}}
\begin{algorithmic}[1]
\scriptsize
\Require Start $S$, Goal $G$, Max iter $M$
\State $T \gets [S]$
\For{$k = 1$ to $M$}
    \State Sample $X_{rand}$
    \State Find nearest $X_{near}$ in $T$
    \State $X_{new} \gets X_{near} + \epsilon \cdot \frac{d}{\|d\|}$
    \If{collision-free}
        \State Add $X_{new}$ to $T$
        \If{$\|X_{new} - G\| < \epsilon$}
            \State \Return path
        \EndIf
    \EndIf
\EndFor
\end{algorithmic}

\subsection{Arc + Cubic Spline}

\begin{frame}[fragile]{Arc Path + Cubic Spline Trajectory}
\begin{columns}
\column{0.5\textwidth}
\textbf{Path Planning: Arc (Fixed)}
\begin{algorithmic}[1]
\scriptsize
\For{$i = 0$ to $N-1$}
    \State $t \gets \frac{i}{N-1}$
    \State $P[i] \gets (1-t)S + tE$
    \State $P[i]_z \gets P[i]_z + 4ht(1-t)$
\EndFor
\end{algorithmic}

\vspace{0.3cm}
\textbf{Trajectory: Cubic Spline}
\begin{algorithmic}[1]
\scriptsize
\Require Times $t_{coarse}$, positions $q_{coarse}$
\For{each joint $j$}
    \State Create CubicSpline with clamped BC
    \State $q_{fine}[:, j] \gets \text{Spline}(t_{fine})$
\EndFor
\State Apply phase unwrapping
\State \Return $q_{fine}$
\end{algorithmic}

\textbf{Properties:} $C^2$ continuous, smooth

\begin{frame}[fragile]{Arc Path + Quintic Polynomial Trajectory}

    \State $P[i]_z \gets P[i]_z + 4ht(1-t)$

\vspace{0.3cm}
\textbf{Trajectory: Quintic Polynomial}
\begin{algorithmic}[1]
\scriptsize
\Require $q_0, q_1, T$
\State $\tau \gets \frac{t}{T}$
\State $q(\tau) = a_0 + a_1\tau + ... + a_5\tau^5$
\State $a_0 = q_0$, $a_1=a_2=0$
\State $a_3=10(q_1-q_0)$
\State $a_4=-15(q_1-q_0)$
\State $a_5=6(q_1-q_0)$
\end{algorithmic}

\textbf{Properties:} $C^3$ continuous, very smooth

    \State $P[i]_z \gets P[i]_z + 4ht(1-t)$
\EndFor
\end{algorithmic}

\vspace{0.3cm}
\textbf{Trajectory: LSPB}
\begin{algorithmic}[1]
\scriptsize
\Require $q_0, q_1, T$, blend time $t_b = 0.3T$
\If{$t < t_b$}
    \State $q = q_0 + \frac{1}{2} a t^2$ \Comment{Accel}
\ElsIf{$t_b \le t \le T - t_b$}
    \State $q = q_0 + v(t - t_b/2)$ \Comment{Const vel}
\Else
    \State $q = q_1 - \frac{1}{2} a(T - t)^2$ \Comment{Decel}
\EndIf
\end{algorithmic}

\textbf{Properties:} $C^1$ continuous, efficient

\section{Conclusion}

\begin{frame}{Key Findings}
\begin{block}{Path Planning}
\begin{itemize}
    \item Choice depends on workspace complexity and obstacle presence
    \item RRT excels in cluttered environments but is computationally expensive
    \item Simple paths (linear, arc) sufficient for open workspaces
\end{itemize}
\end{block}

\begin{block}{Trajectory Planning}
\begin{itemize}
    \item Higher-order polynomials provide smoother motion
    \item Trade-off between smoothness and computational efficiency
    \item Bang-Bang suitable only when high jerk is acceptable
\end{itemize}
\end{block}
